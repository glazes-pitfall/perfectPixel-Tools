```javascript
function snapToGrid(v, gridSize) {
  if (gridSize <= 1) return v;
  return Math.round(v / gridSize) * gridSize;
}

function updateSelectionUI() {
  const hasSelection = !!EditorState.selectionMask;
  const btnDeselect = document.getElementById('btn-deselect');
  const btnInverse  = document.getElementById('btn-inverse');
  if (btnDeselect) btnDeselect.style.display = hasSelection ? '' : 'none';
  if (btnInverse)  btnInverse.style.display  = hasSelection ? '' : 'none';
}

function clearSelection() {
  EditorState.selectionMask = null;
  EditorState.selection = null;
  if (antsRafId) { cancelAnimationFrame(antsRafId); antsRafId = null; }
  _antsPath = null;
  if (selCtx) selCtx.clearRect(0, 0, selCanvas.width, selCanvas.height);
  updateSelectionUI();
}

function computeBoundingBox(mask, W, H) {
  let minX = W, minY = H, maxX = -1, maxY = -1;
  for (let y = 0; y < H; y++) {
    for (let x = 0; x < W; x++) {
      if (mask[x + y * W]) {
        if (x < minX) minX = x; if (y < minY) minY = y;
        if (x > maxX) maxX = x; if (y > maxY) maxY = y;
      }
    }
  }
  if (maxX === -1) return null;
  return { x: minX, y: minY, w: maxX - minX + 1, h: maxY - minY + 1 };
}

function setSelection(mask, bbox) {
  EditorState.selectionMask = mask;
  EditorState.selection = bbox;
  _antsPath = null;  // force Path2D rebuild
  rebuildAntsPath(mask, EditorState.width, EditorState.height);
  updateSelectionUI();
  scheduleAnts();
}

function isSelectedPixel(x, y) {
  if (!EditorState.selectionMask) return true;  // no selection → all pixels writable
  const bb = EditorState.selection;
  if (!bb || x < bb.x || x >= bb.x + bb.w || y < bb.y || y >= bb.y + bb.h) return false;
  return EditorState.selectionMask[x + y * EditorState.width] === 1;
}

function unionMasks(existingMask, newMask, totalPixels) {
  if (!existingMask) return newMask;
  const result = new Uint8Array(totalPixels);
  for (let i = 0; i < totalPixels; i++)
    result[i] = existingMask[i] | newMask[i];
  return result;
}
```

```javascript
function rebuildAntsPath(mask, W, H) {
  _antsPath = new Path2D();
  for (let y = 0; y < H; y++) {
    for (let x = 0; x < W; x++) {
      if (!mask[x + y * W]) continue;
      if (y === 0   || !mask[x + (y-1) * W]) { _antsPath.moveTo(x, y);   _antsPath.lineTo(x+1, y);   }
      if (y === H-1 || !mask[x + (y+1) * W]) { _antsPath.moveTo(x, y+1); _antsPath.lineTo(x+1, y+1); }
      if (x === 0   || !mask[(x-1) + y * W]) { _antsPath.moveTo(x, y);   _antsPath.lineTo(x, y+1);   }
      if (x === W-1 || !mask[(x+1) + y * W]) { _antsPath.moveTo(x+1, y); _antsPath.lineTo(x+1, y+1); }
    }
  }
}

function drawAnts() {
  if (!EditorState.selectionMask || !_antsPath) { antsRafId = null; return; }
  const dpr = window.devicePixelRatio || 1;
  selCtx.clearRect(0, 0, selCanvas.width, selCanvas.height);
  selCtx.save();
  selCtx.scale(dpr, dpr);
  selCtx.lineWidth = 1;
  selCtx.setLineDash([4, 4]);
  selCtx.strokeStyle = '#ffffff';
  selCtx.lineDashOffset = antsDashOffset;
  selCtx.stroke(_antsPath);
  selCtx.strokeStyle = '#000000';
  selCtx.lineDashOffset = antsDashOffset + 4;
  selCtx.stroke(_antsPath);
  selCtx.restore();
  antsDashOffset = (antsDashOffset - 1 + 8) % 8;
  antsRafId = requestAnimationFrame(drawAnts);
}

function scheduleAnts() {
  if (antsRafId) { cancelAnimationFrame(antsRafId); antsRafId = null; }
  antsDashOffset = 0;
  antsRafId = requestAnimationFrame(drawAnts);
}
```

```javascript
// ── Rectangle Marquee tool implementation ────────────────────────────────

let _marqueeStartX = null, _marqueeStartY = null;
let _marqueeCurrentX = null, _marqueeCurrentY = null;

function _marqueeGetSnappedRect(ax, ay, bx, by) {
  const gW = EditorState.gridW || 1;
  const gH = EditorState.gridH || 1;
  const x0 = snapToGrid(ax, gW), y0 = snapToGrid(ay, gH);
  const x1 = snapToGrid(bx, gW), y1 = snapToGrid(by, gH);
  return {
    rx: Math.min(x0, x1), ry: Math.min(y0, y1),
    rw: Math.abs(x1 - x0), rh: Math.abs(y1 - y0)
  };
}

function _marqueeDrawPreview(rx, ry, rw, rh) {
  const dpr = window.devicePixelRatio || 1;
  selCtx.clearRect(0, 0, selCanvas.width, selCanvas.height);
  selCtx.save();
  selCtx.scale(dpr, dpr);
  selCtx.lineWidth = 1;
  selCtx.setLineDash([4, 4]);
  selCtx.strokeStyle = '#ffffff';
  selCtx.lineDashOffset = 0;
  selCtx.strokeRect(rx + 0.5, ry + 0.5, rw, rh);
  selCtx.strokeStyle = '#000000';
  selCtx.lineDashOffset = 4;
  selCtx.strokeRect(rx + 0.5, ry + 0.5, rw, rh);
  selCtx.restore();
}

tools.marquee = {
  onDown(x, y, e) {
    // Clear existing selection unless Shift (Shift-union appends)
    if (!e.shiftKey) {
      // Cancel existing ants but don't call clearSelection — we'll set a new one on up
      if (antsRafId) { cancelAnimationFrame(antsRafId); antsRafId = null; }
      selCtx.clearRect(0, 0, selCanvas.width, selCanvas.height);
    }
    _marqueeStartX = x; _marqueeStartY = y;
    _marqueeCurrentX = x; _marqueeCurrentY = y;
  },
  onMove(x, y) {
    if (_marqueeStartX === null) return;
    _marqueeCurrentX = x; _marqueeCurrentY = y;
    const { rx, ry, rw, rh } = _marqueeGetSnappedRect(
      _marqueeStartX, _marqueeStartY, x, y
    );
    if (rw > 0 && rh > 0) _marqueeDrawPreview(rx, ry, rw, rh);
  },
  onUp(x, y, e) {
    if (_marqueeStartX === null) return;
    const { rx, ry, rw, rh } = _marqueeGetSnappedRect(
      _marqueeStartX, _marqueeStartY, x, y
    );
    _marqueeStartX = null; _marqueeCurrentX = null;

    if (rw === 0 || rh === 0) {
      // Zero-size selection: deselect (a click with no drag)
      clearSelection();
      return;
    }

    // Build mask for snapped rectangle
    const W = EditorState.width, H = EditorState.height;
    const newMask = new Uint8Array(W * H);
    for (let py = ry; py < ry + rh; py++) {
      for (let px = rx; px < rx + rw; px++) {
        if (px >= 0 && px < W && py >= 0 && py < H) {
          newMask[px + py * W] = 1;
        }
      }
    }

    // Shift: union with existing mask; no Shift: replace
    const finalMask = e.shiftKey
      ? unionMasks(EditorState.selectionMask, newMask, W * H)
      : newMask;
    const bbox = computeBoundingBox(finalMask, W, H);
    if (bbox) setSelection(finalMask, bbox);
  },
  onCursor(x, y) {
    // Show crosshair cursor by drawing a minimal cursor indicator
    cursorCtx.clearRect(0, 0, cursorCanvas.width, cursorCanvas.height);
    // Just use CSS cursor:crosshair; no additional canvas drawing needed here
  },
};
```
