2a. transformState 辅助函数：

```javascript
// ── Transform: helper functions ───────────────────────────────────────────

function _getSelCanvasCoords() {
  const caRect  = document.getElementById('canvas-area').getBoundingClientRect();
  const pixRect = pixelCanvas.getBoundingClientRect();
  return {
    originX: pixRect.left - caRect.left,
    originY: pixRect.top  - caRect.top,
    ps:      pixRect.width / EditorState.width,
  };
}

function activateTransform() {
  if (!EditorState.selectionMask || !EditorState.selection) return;
  if (EditorState.transformState) return;  // already active

  const bb   = EditorState.selection;
  const W    = EditorState.width;
  const mask = EditorState.selectionMask;

  // 1. Snapshot originalPixels BEFORE any pixel writes (critical for ESC restore)
  const originalPixels = EditorState.pixels.slice();

  // 2. Extract float pixels from selection bounding box
  const floatPx = new Uint8ClampedArray(bb.w * bb.h * 4);
  for (let fy = 0; fy < bb.h; fy++) {
    for (let fx = 0; fx < bb.w; fx++) {
      const cx = bb.x + fx, cy = bb.y + fy;
      if (mask[cx + cy * W]) {
        const srcI = (cy * W + cx) * 4;
        const dstI = (fy * bb.w + fx) * 4;
        floatPx[dstI]   = EditorState.pixels[srcI];
        floatPx[dstI+1] = EditorState.pixels[srcI+1];
        floatPx[dstI+2] = EditorState.pixels[srcI+2];
        floatPx[dstI+3] = EditorState.pixels[srcI+3];
      }
    }
  }

  // 3. Erase selected pixels from canvas (alpha=0) — AFTER snapshot
  for (let i = 0; i < mask.length; i++) {
    if (mask[i]) {
      const px = i % W, py = (i / W) | 0;
      setPixel(px, py, [0, 0, 0, 0]);
    }
  }
  flushPixels();

  // 4. Stop ants animation
  if (antsRafId) { cancelAnimationFrame(antsRafId); antsRafId = null; }

  // 5. Initialize transformState
  EditorState.transformState = {
    originalPixels,
    floatPixels:      floatPx,
    floatW:           bb.w,
    floatH:           bb.h,
    floatX:           bb.x,
    floatY:           bb.y,
    scaleX:           1.0,
    scaleY:           1.0,
    angleDeg:         0,
    origBbox:         { ...bb },
    _dragMode:        null,
    _dragStartClientX: 0, _dragStartClientY: 0,
    _dragStartFloatX:  0, _dragStartFloatY:  0,
  };

  // 6. Draw static transform UI + show top-bar
  _drawTransformUI();
  _showTransformTopBar();
}

function _drawTransformUI() {
  if (!EditorState.transformState) return;
  const ts  = EditorState.transformState;
  const dpr = window.devicePixelRatio || 1;
  selCtx.clearRect(0, 0, selCanvas.width / dpr, selCanvas.height / dpr);

  const { originX, originY, ps } = _getSelCanvasCoords();
  const sx = originX + ts.floatX * ps;
  const sy = originY + ts.floatY * ps;
  const sw = ts.floatW * ps;
  const sh = ts.floatH * ps;

  // Render float content preview (offscreen canvas → drawImage, safe: source is EditorState pixels)
  if (ts.floatW > 0 && ts.floatH > 0) {
    const off = document.createElement('canvas');
    off.width = ts.floatW; off.height = ts.floatH;
    const offCtx = off.getContext('2d');
    offCtx.putImageData(new ImageData(ts.floatPixels.slice(), ts.floatW, ts.floatH), 0, 0);
    selCtx.imageSmoothingEnabled = false;
    selCtx.drawImage(off, sx, sy, sw, sh);
  }

  // Static dashed border (lineWidth=2, gray-purple, matching Phase 5.1 convention)
  selCtx.strokeStyle = 'rgba(160,150,220,0.9)';
  selCtx.lineWidth   = 2;
  selCtx.setLineDash([4, 3]);
  selCtx.beginPath();
  selCtx.strokeRect(sx - 1, sy - 1, sw + 2, sh + 2);
  selCtx.setLineDash([]);
}

function _showTransformTopBar() {
  const panel = document.getElementById('tool-settings-move');
  if (panel) panel.style.display = 'flex';
  // Sync top-bar inputs to current transformState
  const ts = EditorState.transformState;
  if (!ts) return;
  const sxEl = document.getElementById('opt-scale-x');
  const syEl = document.getElementById('opt-scale-y');
  const aEl  = document.getElementById('opt-rotate-angle');
  if (sxEl) sxEl.value = Math.round(ts.scaleX * 100);
  if (syEl) syEl.value = Math.round(ts.scaleY * 100);
  if (aEl)  aEl.value  = ts.angleDeg;
}

function _hideTransformTopBar() {
  const panel = document.getElementById('tool-settings-move');
  if (panel) panel.style.display = 'none';
}

function _hideDistanceLabel() {
  const label = document.getElementById('transform-distance-label');
  if (label) label.style.display = 'none';
}

function _updateDistanceLabel() {
  const ts = EditorState.transformState;
  if (!ts) return;
  const label = document.getElementById('transform-distance-label');
  if (!label) return;

  const left   = ts.floatX;
  const top    = ts.floatY;
  const right  = EditorState.width  - (ts.floatX + ts.floatW);
  const bottom = EditorState.height - (ts.floatY + ts.floatH);
  label.textContent = `←${left} ↑${top} →${right} ↓${bottom}`;

  const pixRect = pixelCanvas.getBoundingClientRect();
  const ps      = pixRect.width / EditorState.width;
  const screenX = pixRect.left + ts.floatX * ps;
  const screenY = pixRect.top  + (ts.floatY + ts.floatH) * ps + 6;
  label.style.left    = screenX + 'px';
  label.style.top     = screenY + 'px';
  label.style.display = 'block';
}

function applyTransform() {
  const ts = EditorState.transformState;
  if (!ts) return;
  pushHistory();  // one undo step for the entire compound transform

  const W = EditorState.width, H = EditorState.height;
  for (let fy = 0; fy < ts.floatH; fy++) {
    for (let fx = 0; fx < ts.floatW; fx++) {
      const cx = ts.floatX + fx, cy = ts.floatY + fy;
      if (cx < 0 || cx >= W || cy < 0 || cy >= H) continue;  // clip overflow
      const si = (fy * ts.floatW + fx) * 4;
      if (ts.floatPixels[si + 3] === 0) continue;  // transparent → skip
      setPixel(cx, cy, [
        ts.floatPixels[si], ts.floatPixels[si+1],
        ts.floatPixels[si+2], ts.floatPixels[si+3],
      ]);
    }
  }
  flushPixels();
  EditorState.transformState = null;
  clearSelection();
  _hideTransformTopBar();
  _hideDistanceLabel();
}

function cancelTransform() {
  const ts = EditorState.transformState;
  if (!ts) return;
  // Restore original pixels (no history push)
  EditorState.pixels = ts.originalPixels;
  flushPixels();
  EditorState.transformState = null;
  clearSelection();
  _hideTransformTopBar();
  _hideDistanceLabel();
}
```

2b. tools.move 实现（在 tools.wand = {...} 之后赋值）：

```javascript
tools.move = {
  onDown(x, y, e) {
    if (!EditorState.selection) return;  // no selection → no-op
    const ts = EditorState.transformState;
    if (!ts) {
      activateTransform();
      // After activation, start drag immediately
      if (EditorState.transformState) {
        EditorState.transformState._dragMode = 'move';
        EditorState.transformState._dragStartClientX = e.clientX;
        EditorState.transformState._dragStartClientY = e.clientY;
        EditorState.transformState._dragStartFloatX  = EditorState.transformState.floatX;
        EditorState.transformState._dragStartFloatY  = EditorState.transformState.floatY;
      }
    } else {
      // Already in transform mode: start a move drag
      ts._dragMode = 'move';
      ts._dragStartClientX = e.clientX;
      ts._dragStartClientY = e.clientY;
      ts._dragStartFloatX  = ts.floatX;
      ts._dragStartFloatY  = ts.floatY;
    }
  },
  onMove(x, y, e) {
    const ts = EditorState.transformState;
    if (!ts || ts._dragMode !== 'move') return;

    const pixRect = pixelCanvas.getBoundingClientRect();
    const ps      = pixRect.width / EditorState.width;

    const dx = Math.round((e.clientX - ts._dragStartClientX) / ps);
    const dy = Math.round((e.clientY - ts._dragStartClientY) / ps);

    ts.floatX = ts._dragStartFloatX + dx;
    ts.floatY = ts._dragStartFloatY + dy;

    _drawTransformUI();
    _updateDistanceLabel();
  },
  onUp(x, y, e) {
    const ts = EditorState.transformState;
    if (!ts) return;
    ts._dragMode = null;
    _updateDistanceLabel();
  },
  onCursor(x, y) {
    // Move cursor handled by CSS 'move' in setActiveTool
  },
};
```
