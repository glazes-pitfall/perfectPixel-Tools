**3. 在 `<script>` 内（DOMContentLoaded 之前，全局作用域）添加下载相关函数：**
```javascript
// ── 下载弹窗 ──────────────────────────────────────────────────────────────
let _lastBlobUrl = null;  // 维护以便在下次下载/关闭弹窗时 revoke

function triggerDownload(scale) {
  if (!EditorState.pixels || !EditorState.width) return;
  const newW = EditorState.width  * scale;
  const newH = EditorState.height * scale;

  // 从 EditorState.pixels 构建源 canvas（绕开 premultiplied alpha）
  const src = document.createElement('canvas');
  src.width = EditorState.width; src.height = EditorState.height;
  src.getContext('2d').putImageData(
    new ImageData(EditorState.pixels.slice(), EditorState.width, EditorState.height), 0, 0
  );

  // 整数倍缩放（最近邻，保持像素艺术锐利边缘）
  const off = document.createElement('canvas');
  off.width = newW; off.height = newH;
  const dctx = off.getContext('2d');
  dctx.imageSmoothingEnabled = false;
  dctx.drawImage(src, 0, 0, newW, newH);

  off.toBlob(blob => {
    // Revoke 旧 Blob URL
    if (_lastBlobUrl) { URL.revokeObjectURL(_lastBlobUrl); _lastBlobUrl = null; }

    const blobUrl = URL.createObjectURL(blob);
    _lastBlobUrl = blobUrl;

    // 显示预览
    const previewScale = Math.min(1, 480 / Math.max(newW, newH));
    const previewImg = document.getElementById('dl-preview-img');
    previewImg.src = blobUrl;
    previewImg.style.width  = Math.round(newW * previewScale) + 'px';
    previewImg.style.height = Math.round(newH * previewScale) + 'px';
    document.getElementById('dl-size-info').textContent = `${newW} × ${newH} px`;
    document.getElementById('dl-preview').style.display = 'block';

    // 触发下载
    const baseName = (EditorState.filename || '').replace(/\.[^.]+$/, '') || 'output';
    const filename = `${baseName}_pixelated_${scale}x.png`;
    const a = document.createElement('a');
    a.href = blobUrl; a.download = filename;
    document.body.appendChild(a); a.click(); document.body.removeChild(a);
    // 不立即 revoke：用户可能右键预览图另存为
  }, 'image/png');
}

function goHome() {
  if (EditorState.pixels) {
    if (!confirm('将丢失当前进度，是否继续？')) return;
  }
  window.location.href = '/';
}

function openDownloadModal() {
  document.getElementById('dl-preview').style.display = 'none';
  document.getElementById('dl-scale-slider').value = 1;
  document.getElementById('dl-scale-num').value = 1;
  const modal = document.getElementById('download-modal');
  modal.style.display = 'flex';
}

function closeDownloadModal() {
  document.getElementById('download-modal').style.display = 'none';
  // 关闭时 revoke blob URL（预览图不再需要）
  if (_lastBlobUrl) { URL.revokeObjectURL(_lastBlobUrl); _lastBlobUrl = null; }
}
```

// ── 下载弹窗事件绑定 ───────────────────────────────────────────────────────
document.getElementById('btn-download-open').addEventListener('click', openDownloadModal);
document.getElementById('dl-btn-cancel').addEventListener('click', closeDownloadModal);

  const scale = Math.max(1, Math.min(100, parseInt(document.getElementById('dl-scale-num').value) || 1));

// 滑条与数值输入框同步
const dlSlider = document.getElementById('dl-scale-slider');
const dlNum    = document.getElementById('dl-scale-num');
dlSlider.addEventListener('input', () => { dlNum.value = dlSlider.value; });
dlNum.addEventListener('input', () => {
  const v = Math.max(1, Math.min(100, parseInt(dlNum.value) || 1));
  dlSlider.value = v;
});

// 点击遮罩背景关闭弹窗
document.getElementById('download-modal').addEventListener('click', e => {
  if (e.target === document.getElementById('download-modal')) closeDownloadModal();
});
