```javascript
// ── Canvas Size Tool ──────────────────────────────────────────────────────
function _getCanvasSizeParams() {
  const W  = parseInt(document.getElementById('cfg-width').value)  || EditorState.width;
  const H  = parseInt(document.getElementById('cfg-height').value) || EditorState.height;
  const L  = parseInt(document.getElementById('cfg-left').value)   || 0;
  const R  = parseInt(document.getElementById('cfg-right').value)  || 0;
  const T  = parseInt(document.getElementById('cfg-top').value)    || 0;
  const B  = parseInt(document.getElementById('cfg-bottom').value) || 0;
  // 最终新画布尺寸 = W/H + L+R/T+B（L/R/T/B 在 W/H 基础上叠加偏移）
  const newW = Math.max(1, W + L + R);
  const newH = Math.max(1, H + T + B);
  const offsetL = L;  // 旧内容向右移动 L 像素
  const offsetT = T;  // 旧内容向下移动 T 像素
  return { newW, newH, offsetL, offsetT };
}

function drawCanvasSizeGuides() {
  if (EditorState.activeTool !== 'canvas-size') return;
  const dpr = window.devicePixelRatio || 1;
  selCtx.clearRect(0, 0, selCanvas.width / dpr, selCanvas.height / dpr);

  if (!EditorState.pixels) return;

  const { newW, newH, offsetL, offsetT } = _getCanvasSizeParams();

  // 坐标转换：画布像素 → selCanvas CSS 坐标
  const pixRect = pixelCanvas.getBoundingClientRect();
  const caRect  = document.getElementById('canvas-area').getBoundingClientRect();
  const originX = pixRect.left - caRect.left;
  const originY = pixRect.top  - caRect.top;
  const ps      = pixRect.width / EditorState.width;  // CSS px per canvas pixel

  // 新画布四边在 selCanvas CSS 坐标中的位置（offsetL/T = 旧内容偏移量，新画布左上角在 -offsetL/-offsetT）
  const left   = originX - offsetL * ps;
  const top    = originY - offsetT * ps;
  const right  = left + newW * ps;
  const bottom = top  + newH * ps;

  // 扩张区蓝色棋盘格叠加（半透明蓝矩形）
  const caW = selCanvas.width / dpr;
  const caH = selCanvas.height / dpr;
  selCtx.save();
  selCtx.fillStyle = 'rgba(100,140,220,0.15)';
  // 上扩张区
  if (top < originY)
    selCtx.fillRect(left, top, newW * ps, originY - top);
  // 下扩张区
  if (bottom > originY + EditorState.height * ps)
    selCtx.fillRect(left, originY + EditorState.height * ps, newW * ps, bottom - (originY + EditorState.height * ps));
  // 左扩张区
  if (left < originX)
    selCtx.fillRect(left, top, originX - left, newH * ps);
  // 右扩张区
  if (right > originX + EditorState.width * ps)
    selCtx.fillRect(originX + EditorState.width * ps, top, right - (originX + EditorState.width * ps), newH * ps);
  selCtx.restore();

  // 紫色实线参考线（跨越整个 canvas-area）
  selCtx.save();
  selCtx.strokeStyle = '#7c6af7';
  selCtx.lineWidth = 1;
  selCtx.setLineDash([]);
  // 左边线
  selCtx.beginPath(); selCtx.moveTo(left, 0); selCtx.lineTo(left, caH); selCtx.stroke();
  // 右边线
  selCtx.beginPath(); selCtx.moveTo(right, 0); selCtx.lineTo(right, caH); selCtx.stroke();
  // 上边线
  selCtx.beginPath(); selCtx.moveTo(0, top); selCtx.lineTo(caW, top); selCtx.stroke();
  // 下边线
  selCtx.beginPath(); selCtx.moveTo(0, bottom); selCtx.lineTo(caW, bottom); selCtx.stroke();
  selCtx.restore();
}
```

**2. applyCanvasSize 函数（紧随 drawCanvasSizeGuides 之后）：**
```javascript
function applyCanvasSize() {
  if (!EditorState.pixels) return;
  const { newW, newH, offsetL, offsetT } = _getCanvasSizeParams();
  const oldPixels = EditorState.pixels;
  const oldW = EditorState.width;
  const oldH = EditorState.height;

  pushHistory();  // save-before（Apply 是即时操作）

  const newPixels = new Uint8ClampedArray(newW * newH * 4);  // 全透明

  const srcX0 = Math.max(0, -offsetL);
  const srcY0 = Math.max(0, -offsetT);
  const dstX0 = Math.max(0, offsetL);
  const dstY0 = Math.max(0, offsetT);
  const copyW = Math.min(oldW - srcX0, newW - dstX0);
  const copyH = Math.min(oldH - srcY0, newH - dstY0);

  if (copyW > 0 && copyH > 0) {
    for (let row = 0; row < copyH; row++) {
      const srcOff = ((srcY0 + row) * oldW + srcX0) * 4;
      const dstOff = ((dstY0 + row) * newW + dstX0) * 4;
      newPixels.set(oldPixels.subarray(srcOff, srcOff + copyW * 4), dstOff);
    }
  }

  EditorState.pixels = newPixels;
  clearSelection();          // 清除旧选区 mask（坐标系已失效）
  initCanvases(newW, newH);  // 重建三层 canvas + scroll 空间
  flushPixels();
  setActiveTool('pencil');   // 退出 canvas-size 模式
}
```
