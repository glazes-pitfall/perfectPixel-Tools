```javascript
async function loadImageFromB64(b64) {
  const blob   = await (await fetch('data:image/png;base64,' + b64)).blob();
  const bitmap = await createImageBitmap(blob);
  const { width, height } = bitmap;
  initCanvases(width, height);
  pixelCtx.drawImage(bitmap, 0, 0);
  const imageData = pixelCtx.getImageData(0, 0, width, height);
  EditorState.pixels = imageData.data.slice();
  clearSelection();
  flushPixels();
  bitmap.close();
}
```

**3. 新增 showDropZone 和 handleFileUpload 函数（在 loadImageFromB64 之后）：**
```javascript
function showDropZone() {
  const canvasArea = document.getElementById('canvas-area');
  const zone = document.createElement('div');
  zone.id = 'drop-zone';
  zone.style.cssText = 'position:absolute;inset:0;z-index:100;display:flex;align-items:center;justify-content:center;flex-direction:column;gap:12px;cursor:pointer;background:var(--surface);';
  zone.innerHTML = '<div style="font-size:48px;">🖼️</div><div style="font-size:16px;color:var(--text-muted);">点击或拖拽图片到此处</div><div style="font-size:12px;color:var(--text-muted);">支持 PNG / JPG / WEBP 等格式</div>';
  canvasArea.appendChild(zone);

  const fileInput = document.createElement('input');
  fileInput.type = 'file'; fileInput.accept = 'image/*'; fileInput.style.display = 'none';
  canvasArea.appendChild(fileInput);

  zone.addEventListener('click', () => fileInput.click());
  fileInput.addEventListener('change', () => {
    if (fileInput.files[0]) handleFileUpload(fileInput.files[0]);
  });
  zone.addEventListener('dragover', e => { e.preventDefault(); zone.style.background = 'rgba(124,106,247,.08)'; });
  zone.addEventListener('dragleave', () => { zone.style.background = 'var(--surface)'; });
  zone.addEventListener('drop', e => {
    e.preventDefault();
    const f = e.dataTransfer.files[0];
    if (f && f.type.startsWith('image/')) handleFileUpload(f);
  });
}

async function handleFileUpload(file) {
  EditorState.filename = file.name;
  const reader = new FileReader();
  reader.onload = async e => {
    const b64 = e.target.result.replace(/^data:image\/\w+;base64,/, '');
    document.getElementById('drop-zone')?.remove();
    await loadImageFromB64(b64);
    pushHistory();
    bindPostLoadEvents();
  };
  reader.readAsDataURL(file);
}
```

**4. 将 pixel inspector 事件绑定提取为 bindPostLoadEvents 函数：**
将现有 `loadPlaceholderImage().then(() => { ... })` 回调内的 pixel inspector 代码（cursorCanvas pointermove/pointerleave 两个监听器）提取为独立函数 `bindPostLoadEvents()`，放在 `showDropZone` 之后：
```javascript
let _postLoadEventsBound = false;
function bindPostLoadEvents() {
  if (_postLoadEventsBound) return;  // 防重复绑定
  _postLoadEventsBound = true;
  // pixel inspector（pointermove）
  cursorCanvas.addEventListener('pointermove', e => {
    if (!EditorState.pixels) return;
    const [cx, cy] = viewportToCanvas(e.clientX, e.clientY);
    const [r, g, b, a] = getPixel(cx, cy);
    document.getElementById('insp-x').textContent = cx;
    document.getElementById('insp-y').textContent = cy;
    document.getElementById('insp-r').textContent = r;
    document.getElementById('insp-g').textContent = g;
    document.getElementById('insp-b').textContent = b;
    document.getElementById('insp-a').textContent = a;
    document.getElementById('insp-swatch').style.background =
      `rgba(${r},${g},${b},${a / 255})`;
  });
  cursorCanvas.addEventListener('pointerleave', () => {
    ['insp-x', 'insp-y', 'insp-r', 'insp-g', 'insp-b', 'insp-a'].forEach(id => {
      document.getElementById(id).textContent = '—';
    });
    document.getElementById('insp-swatch').style.background = 'transparent';
  });
}
```
