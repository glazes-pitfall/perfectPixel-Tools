// HSL <-> RGB conversion (no external library)
function hslToRgb(h, s, l) {
  s /= 100; l /= 100;
  const k = n => (n + h / 30) % 12;
  const a = s * Math.min(l, 1 - l);
  const f = n => l - a * Math.max(-1, Math.min(k(n) - 3, Math.min(9 - k(n), 1)));
  return [Math.round(f(0) * 255), Math.round(f(8) * 255), Math.round(f(4) * 255)];
}

function rgbToHsl(r, g, b) {
  r /= 255; g /= 255; b /= 255;
  const max = Math.max(r, g, b), min = Math.min(r, g, b);
  const l = (max + min) / 2;
  if (max === min) return [0, 0, Math.round(l * 100)];
  const d = max - min;
  const s = l > 0.5 ? d / (2 - max - min) : d / (max + min);
  let h;
  if (max === r) h = ((g - b) / d + (g < b ? 6 : 0)) / 6;
  else if (max === g) h = ((b - r) / d + 2) / 6;
  else h = ((r - g) / d + 4) / 6;
  return [Math.round(h * 360), Math.round(s * 100), Math.round(l * 100)];
}

// Color picker state
let pickerCanvas = null, pickerCtx = null;
let currentHue = 0, currentSat = 0, currentLit = 0;
const PICKER_CX = 80, PICKER_CY = 80;
const RING_OUTER = 75, RING_INNER = 55;
const SQ_X = 35, SQ_Y = 35, SQ_W = 90, SQ_H = 90;
let _pickerDragZone = null;  // 'ring' | 'square' | null

function drawHueRing(ctx) {
  const grad = ctx.createConicGradient(-Math.PI / 2, PICKER_CX, PICKER_CY);
  const stops = [
    [0, '#ff0000'], [1/6, '#ffff00'], [2/6, '#00ff00'],
    [3/6, '#00ffff'], [4/6, '#0000ff'], [5/6, '#ff00ff'], [1, '#ff0000']
  ];
  stops.forEach(([pos, color]) => grad.addColorStop(pos, color));
  ctx.beginPath();
  ctx.arc(PICKER_CX, PICKER_CY, RING_OUTER, 0, 2 * Math.PI);
  ctx.arc(PICKER_CX, PICKER_CY, RING_INNER, 0, 2 * Math.PI, true);
  ctx.fillStyle = grad;
  ctx.fill();
}

function drawSLSquare(ctx) {
  // Pass 1: saturation (gray at 0% -> pure hue at 100%)
  const satGrad = ctx.createLinearGradient(SQ_X, SQ_Y, SQ_X + SQ_W, SQ_Y);
  satGrad.addColorStop(0, 'hsl(' + currentHue + ', 0%, 50%)');
  satGrad.addColorStop(1, 'hsl(' + currentHue + ', 100%, 50%)');
  ctx.fillStyle = satGrad;
  ctx.fillRect(SQ_X, SQ_Y, SQ_W, SQ_H);
  // Pass 2: lightness overlay (white top -> transparent middle -> black bottom)
  const litGrad = ctx.createLinearGradient(SQ_X, SQ_Y, SQ_X, SQ_Y + SQ_H);
  litGrad.addColorStop(0, 'rgba(255,255,255,1)');
  litGrad.addColorStop(0.5, 'rgba(255,255,255,0)');
  litGrad.addColorStop(0.5, 'rgba(0,0,0,0)');
  litGrad.addColorStop(1, 'rgba(0,0,0,1)');
  ctx.fillStyle = litGrad;
  ctx.fillRect(SQ_X, SQ_Y, SQ_W, SQ_H);
}

function drawPickerIndicators(ctx) {
  // Ring indicator: small white circle on hue ring at currentHue
  const ringR = (RING_OUTER + RING_INNER) / 2;
  const hueRad = (currentHue - 90) * Math.PI / 180;
  const rx = PICKER_CX + ringR * Math.cos(hueRad);
  const ry = PICKER_CY + ringR * Math.sin(hueRad);
  ctx.beginPath();
  ctx.arc(rx, ry, 5, 0, 2 * Math.PI);
  ctx.fillStyle = '#fff';
  ctx.fill();
  ctx.strokeStyle = '#000';
  ctx.lineWidth = 1;
  ctx.stroke();

  // SL square indicator: small circle at current S/L position
  const sx = SQ_X + (currentSat / 100) * SQ_W;
  // Lightness: 100% at top, 0% at bottom — map lightness to y inverted
  const sy = SQ_Y + (1 - currentLit / 100) * SQ_H;
  ctx.beginPath();
  ctx.arc(sx, sy, 5, 0, 2 * Math.PI);
  ctx.fillStyle = 'rgba(255,255,255,0.3)';
  ctx.fill();
  ctx.strokeStyle = '#fff';
  ctx.lineWidth = 1.5;
  ctx.stroke();
}

function redrawPicker() {
  if (!pickerCtx) return;
  pickerCtx.clearRect(0, 0, 160, 160);
  drawHueRing(pickerCtx);
  drawSLSquare(pickerCtx);
  drawPickerIndicators(pickerCtx);
}

// Single sync path: any control change -> EditorState.foregroundColor -> syncColorUI()
let _syncLock = false;
function syncColorUI() {
  if (_syncLock) return;
  _syncLock = true;
  const [r, g, b] = EditorState.foregroundColor;
  // Update hex input
  const hex = '#' + [r, g, b].map(v => v.toString(16).padStart(2, '0')).join('');
  const hexEl = document.getElementById('clr-hex');
  if (hexEl) hexEl.value = hex;
  // Update RGB inputs
  const rEl = document.getElementById('clr-r'), gEl = document.getElementById('clr-g'), bEl = document.getElementById('clr-b');
  if (rEl) rEl.value = r;
  if (gEl) gEl.value = g;
  if (bEl) bEl.value = b;
  // Update swatch
  const swatch = document.getElementById('clr-swatch');
  if (swatch) swatch.style.background = 'rgb(' + r + ',' + g + ',' + b + ')';
  // Update HSL state and redraw picker
  [currentHue, currentSat, currentLit] = rgbToHsl(r, g, b);
  redrawPicker();
  _syncLock = false;
}

tools.eyedropper = {
  onDown(x, y) {
    const [r, g, b, a] = getPixel(x, y);
    if (a === 0) {
      // Transparent pixel: do NOT change foreground color
      // (already showing transparent indicator from onCursor)
      return;
    }
    EditorState.foregroundColor = [r, g, b, 255];
    syncColorUI();
    // Restore previous tool
    setActiveTool(EditorState._prevTool || 'pencil');
    // Clear eyedropper UI
    const hoverSwatch = document.getElementById('eyedropper-hover-swatch');
    const transparentLabel = document.getElementById('eyedropper-transparent-label');
    if (hoverSwatch) hoverSwatch.style.display = 'none';
    if (transparentLabel) transparentLabel.style.display = 'none';
  },
  onMove() {},
  onUp() {},
  onCursor(x, y) {
    // Draw crosshair on cursor-canvas
    cursorCtx.clearRect(0, 0, cursorCanvas.width, cursorCanvas.height);
    cursorCtx.strokeStyle = 'rgba(255,255,255,0.9)';
    cursorCtx.lineWidth = 1 / (window.devicePixelRatio || 1);
    // Vertical line
    cursorCtx.beginPath();
    cursorCtx.moveTo(x, Math.max(0, y - 3));
    cursorCtx.lineTo(x, Math.min(EditorState.height, y + 4));
    cursorCtx.stroke();
    // Horizontal line
    cursorCtx.beginPath();
    cursorCtx.moveTo(Math.max(0, x - 3), y);
    cursorCtx.lineTo(Math.min(EditorState.width, x + 4), y);
    cursorCtx.stroke();

    // Update hover swatch in picker panel
    const [r, g, b, a] = getPixel(x, y);
    const hoverSwatch = document.getElementById('eyedropper-hover-swatch');
    const transparentLabel = document.getElementById('eyedropper-transparent-label');
    if (a === 0) {
      if (hoverSwatch) hoverSwatch.style.display = 'none';
      if (transparentLabel) transparentLabel.style.display = 'inline';
    } else {
      if (hoverSwatch) {
        hoverSwatch.style.display = 'block';
        hoverSwatch.style.background = 'rgb(' + r + ',' + g + ',' + b + ')';
      }
      if (transparentLabel) transparentLabel.style.display = 'none';
    }
  },
};
