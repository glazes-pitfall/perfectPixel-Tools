```javascript
// ── History (undo/redo) ───────────────────────────────────────────────────

let btnUndo = null;
let btnRedo = null;

function initHistoryButtons() {
  btnUndo = document.getElementById('btn-undo');
  btnRedo = document.getElementById('btn-redo');
  btnUndo.addEventListener('click', undo);
  btnRedo.addEventListener('click', redo);
  updateHistoryButtons();
}

function updateHistoryButtons() {
  if (!btnUndo || !btnRedo) return;
  // canUndo: historyIndex >= 1 (index 0 is the initial snapshot, can go back to nothing)
  // 精确规则：historyIndex > 0 表示有"之前"的快照可恢复
  btnUndo.disabled = EditorState.historyIndex <= 0;
  // canRedo: historyIndex < history.length - 1
  btnRedo.disabled = EditorState.historyIndex >= EditorState.history.length - 1;
}

/**
 * 保存当前 EditorState.pixels 快照到 history stack。
 * 在 pointerdown 时调用，每次操作只调用一次（不在 pointermove 中调用）。
 */
function pushHistory() {
  // 截断 redo 分支
  EditorState.history.splice(EditorState.historyIndex + 1);
  // 压入当前快照
  EditorState.history.push(EditorState.pixels.slice());
  // overflow：保持 stack 不超过 MAX_HISTORY
  if (EditorState.history.length > EditorState.MAX_HISTORY) {
    EditorState.history.shift();
    // historyIndex 保持指向末尾（shift 后长度 = MAX_HISTORY，index = MAX_HISTORY - 1）
  } else {
    EditorState.historyIndex++;
  }
  updateHistoryButtons();
}

function undo() {
  if (EditorState.historyIndex <= 0) return;
  EditorState.historyIndex--;
  EditorState.pixels = EditorState.history[EditorState.historyIndex].slice();
  flushPixels();
  updateHistoryButtons();
}

function redo() {
  if (EditorState.historyIndex >= EditorState.history.length - 1) return;
  EditorState.historyIndex++;
  EditorState.pixels = EditorState.history[EditorState.historyIndex].slice();
  flushPixels();
  updateHistoryButtons();
}
```

**重要逻辑说明：**
- `pushHistory()` 第一次调用时：historyIndex 从 -1 → 0，history[0] = 当前快照（代表"最早状态"）
- `undo()` 需要 historyIndex > 0（即有"更早"的快照）；undo 后 historyIndex-- 指向上一快照
- 因此正确条件是：canUndo = historyIndex > 0（不是 >= 0）
