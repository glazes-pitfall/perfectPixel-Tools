**2a. scaleNearestNeighbor（Plan 06 核心算法）**

```javascript
function scaleNearestNeighbor(pixels, srcW, srcH, dstW, dstH) {
  const out = new Uint8ClampedArray(dstW * dstH * 4);
  const xRatio = srcW / dstW;
  const yRatio = srcH / dstH;
  for (let dy = 0; dy < dstH; dy++) {
    const sy = Math.min(srcH - 1, Math.floor(dy * yRatio));
    for (let dx = 0; dx < dstW; dx++) {
      const sx = Math.min(srcW - 1, Math.floor(dx * xRatio));
      const srcI = (sy * srcW + sx) * 4;
      const dstI = (dy * dstW + dx) * 4;
      out[dstI]   = pixels[srcI];
      out[dstI+1] = pixels[srcI+1];
      out[dstI+2] = pixels[srcI+2];
      out[dstI+3] = pixels[srcI+3];
    }
  }
  return out;
}
```

**2c. hitTestHandle**

```javascript
function hitTestHandle(clientX, clientY) {
  if (!EditorState.transformState) return null;
  const ts = EditorState.transformState;
  const caRect  = document.getElementById('canvas-area').getBoundingClientRect();
  const pixRect = pixelCanvas.getBoundingClientRect();
  const originX = pixRect.left - caRect.left;
  const originY = pixRect.top  - caRect.top;
  const ps      = pixRect.width / EditorState.width;

  const lx = clientX - caRect.left;
  const ly = clientY - caRect.top;

  const sx = originX + ts.floatX * ps;
  const sy = originY + ts.floatY * ps;
  const sw = ts.floatW * ps;
  const sh = ts.floatH * ps;
  const HALF = 6;  // slightly larger hit zone than visual (12px) for usability

  const handlePositions = [
    [sx,        sy       ],  // 0 TL
    [sx + sw/2, sy       ],  // 1 TC
    [sx + sw,   sy       ],  // 2 TR
    [sx,        sy + sh/2],  // 3 ML
    [sx + sw,   sy + sh/2],  // 4 MR
    [sx,        sy + sh  ],  // 5 BL
    [sx + sw/2, sy + sh  ],  // 6 BC
    [sx + sw,   sy + sh  ],  // 7 BR
  ];
  for (let i = 0; i < handlePositions.length; i++) {
    const [hx, hy] = handlePositions[i];
    if (Math.abs(lx - hx) <= HALF && Math.abs(ly - hy) <= HALF) return i;
  }
  return null;
}
```

onDown 修改（在已有 'move' drag 逻辑之前，先检查 handle hit-test）：
```javascript
onDown(x, y, e) {
  // Case 1: transform already active → check handle hit-test first
  const existingTs = EditorState.transformState;
  if (existingTs) {
    const handleIdx = hitTestHandle(e.clientX, e.clientY);
    if (handleIdx !== null) {
      // Store original floatPixels for scale operations (source of truth)
      existingTs._origFloatPixels = existingTs.floatPixels.slice();
      existingTs._origFloatW      = existingTs.floatW;
      existingTs._origFloatH      = existingTs.floatH;
      existingTs._dragMode        = 'handle-' + handleIdx;
      existingTs._dragStartClientX = e.clientX;
      existingTs._dragStartClientY = e.clientY;
      existingTs._dragStartScaleX  = existingTs.scaleX;
      existingTs._dragStartScaleY  = existingTs.scaleY;
      existingTs._dragStartFloatX  = existingTs.floatX;
      existingTs._dragStartFloatY  = existingTs.floatY;
      return;
    }
    // No handle hit → treat as move drag (existing behavior)
    existingTs._dragMode        = 'move';
    existingTs._dragStartClientX = e.clientX;
    existingTs._dragStartClientY = e.clientY;
    existingTs._dragStartFloatX  = existingTs.floatX;
    existingTs._dragStartFloatY  = existingTs.floatY;
    return;
  }
  // Case 2: no transform yet → activate
  if (!EditorState.selection) return;
  activateTransform();
  if (EditorState.transformState) {
    EditorState.transformState._dragMode        = 'move';
    EditorState.transformState._dragStartClientX = e.clientX;
    EditorState.transformState._dragStartClientY = e.clientY;
    EditorState.transformState._dragStartFloatX  = EditorState.transformState.floatX;
    EditorState.transformState._dragStartFloatY  = EditorState.transformState.floatY;
  }
},
```

onMove 修改（在已有 'move' 分支之后，添加 'handle-N' 分支）：
```javascript
onMove(x, y, e) {
  const ts = EditorState.transformState;
  if (!ts) return;

  if (ts._dragMode === 'move') {
    // (existing move drag logic from Plan 01)
    const pixRect = pixelCanvas.getBoundingClientRect();
    const ps      = pixRect.width / EditorState.width;
    const dx = Math.round((e.clientX - ts._dragStartClientX) / ps);
    const dy = Math.round((e.clientY - ts._dragStartClientY) / ps);
    ts.floatX = ts._dragStartFloatX + dx;
    ts.floatY = ts._dragStartFloatY + dy;
    _drawTransformUI();
    _updateDistanceLabel();
    return;
  }

  if (ts._dragMode && ts._dragMode.startsWith('handle-')) {
    const handleIdx = parseInt(ts._dragMode.split('-')[1], 10);
    const pixRect   = pixelCanvas.getBoundingClientRect();
    const ps        = pixRect.width / EditorState.width;

    // Delta in canvas pixels
    const dxPx = Math.round((e.clientX - ts._dragStartClientX) / ps);
    const dyPx = Math.round((e.clientY - ts._dragStartClientY) / ps);

    // Original bbox dimensions
    const origW = ts.origBbox.w;
    const origH = ts.origBbox.h;
    const lockAspect = document.getElementById('opt-scale-lock')?.checked ?? true;

    // Compute new scale based on which handle is dragged
    // Corners: both axes; TC/BC: Y only; ML/MR: X only
    let newScaleX = ts._dragStartScaleX;
    let newScaleY = ts._dragStartScaleY;

    const corners   = [0, 2, 5, 7];  // TL TR BL BR
    const yOnlyH    = [1, 6];         // TC BC
    const xOnlyH    = [3, 4];         // ML MR

    if (corners.includes(handleIdx)) {
      // For right-side corners (2,4,7): drag right = increase X
      // For left-side corners (0,3,5): drag right = decrease X
      const xSign = [2, 4, 7].includes(handleIdx) ? 1 : -1;
      const ySign = [5, 6, 7].includes(handleIdx) ? 1 : -1;
      newScaleX = Math.max(0.0625, (origW * ts._dragStartScaleX + dxPx * xSign) / origW);
      newScaleY = Math.max(0.0625, (origH * ts._dragStartScaleY + dyPx * ySign) / origH);
      if (lockAspect) {
        // Use the axis with larger relative change
        const relX = Math.abs(newScaleX - ts._dragStartScaleX);
        const relY = Math.abs(newScaleY - ts._dragStartScaleY);
        if (relX > relY) newScaleY = newScaleX;
        else             newScaleX = newScaleY;
      }
    } else if (yOnlyH.includes(handleIdx)) {
      const ySign = handleIdx === 6 ? 1 : -1;
      newScaleY = Math.max(0.0625, (origH * ts._dragStartScaleY + dyPx * ySign) / origH);
      if (lockAspect) newScaleX = newScaleY;
    } else if (xOnlyH.includes(handleIdx)) {
      const xSign = handleIdx === 4 ? 1 : -1;
      newScaleX = Math.max(0.0625, (origW * ts._dragStartScaleX + dxPx * xSign) / origW);
      if (lockAspect) newScaleY = newScaleX;
    }

    ts.scaleX = newScaleX;
    ts.scaleY = newScaleY;

    // Resample from _origFloatPixels (the original captured pixels, not from last frame)
    const newW = Math.max(1, Math.round(ts.origBbox.w * ts.scaleX));
    const newH = Math.max(1, Math.round(ts.origBbox.h * ts.scaleY));
    ts.floatPixels = scaleNearestNeighbor(ts._origFloatPixels, ts._origFloatW, ts._origFloatH, newW, newH);
    ts.floatW = newW;
    ts.floatH = newH;

    // Sync top-bar inputs
    const sxEl = document.getElementById('opt-scale-x');
    const syEl = document.getElementById('opt-scale-y');
    if (sxEl) sxEl.value = Math.round(ts.scaleX * 100);
    if (syEl) syEl.value = Math.round(ts.scaleY * 100);

    _drawTransformUI();
    return;
  }
},
```

```javascript
// ── Transform top-bar input bindings ─────────────────────────────────────
let _scaleInputTimer = null;

function _applyScaleFromInputs() {
  const ts = EditorState.transformState;
  if (!ts) return;

  const sxVal = parseFloat(document.getElementById('opt-scale-x')?.value) || 100;
  const syVal = parseFloat(document.getElementById('opt-scale-y')?.value) || 100;

  ts.scaleX = Math.max(0.0625, Math.min(16, sxVal / 100));
  ts.scaleY = Math.max(0.0625, Math.min(16, syVal / 100));

  // Resample from _origFloatPixels (snapshot at activation)
  // If _origFloatPixels not yet stored (no handle drag happened), use floatPixels at activation
  const srcPx = ts._origFloatPixels || ts.floatPixels;
  const srcW  = ts._origFloatW      || ts.origBbox.w;
  const srcH  = ts._origFloatH      || ts.origBbox.h;

  const newW = Math.max(1, Math.round(ts.origBbox.w * ts.scaleX));
  const newH = Math.max(1, Math.round(ts.origBbox.h * ts.scaleY));

  ts.floatPixels = scaleNearestNeighbor(srcPx, srcW, srcH, newW, newH);
  ts.floatW      = newW;
  ts.floatH      = newH;

  _drawTransformUI();
}

const scaleXInput   = document.getElementById('opt-scale-x');
const scaleYInput   = document.getElementById('opt-scale-y');
const scaleLockChk  = document.getElementById('opt-scale-lock');

if (scaleXInput) {
  scaleXInput.addEventListener('input', () => {
    if (scaleLockChk?.checked) {
      if (scaleYInput) scaleYInput.value = scaleXInput.value;
    }
    clearTimeout(_scaleInputTimer);
    _scaleInputTimer = setTimeout(_applyScaleFromInputs, 300);
  });
}
if (scaleYInput) {
  scaleYInput.addEventListener('input', () => {
    if (scaleLockChk?.checked) {
      if (scaleXInput) scaleXInput.value = scaleYInput.value;
    }
    clearTimeout(_scaleInputTimer);
    _scaleInputTimer = setTimeout(_applyScaleFromInputs, 300);
  });
}
```
